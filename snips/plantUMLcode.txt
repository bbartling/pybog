@startuml
title Bog Maker - Tiered AI Agent Process

' Style customizations
skinparam activity {
StartColor #22c55e
StopColor #ef4444
BorderColor #16a34a
BackgroundColor #f0fdf4
ArrowColor #333
}
skinparam partition {
BorderColor #a1a1aa
BackgroundColor #f4f4f5
}
skinparam note {
BackgroundColor #feFce8
BorderColor #eab308
}

start

partition "API Request Handling" {
:Receive user request
(description, filename, etc.);
:Check user rate limit;
:Create temporary directory;
}

while (Attempt < max_iters?) is (Yes)
partition "Main Agent Loop" {
if (Attempt = 1?) then (Yes)
:Generate Script
(using <b>Flash</b> model);
else (No)
if (Attempt = Escalation Point?) then (Yes)
partition "Escalation" #fef2e2 {
:⚠️ ESCALATE
Fix Script
(using <b>Pro</b> model);
}
else (No)
:Fix Script
(using <b>Flash</b> model);
endif
endif
:Write script to temp folder;
:Run script in subprocess;
}

if (Run OK AND .bog file created?) then (Yes)
    break
else (No)
    :Capture stderr/stdout;
    :Increment attempt;
endif

endwhile (is No)

if (BOG File Exists?) then (Yes)
partition "Quality Assurance (QA)" {
if (Last attempt used <b>Pro</b> model?) then (No)
#fefce8:-> Pro Verification Pass
:<b>Pro</b> model reviews script;
if (Refinements suggested?) then (Yes)
:Write new script;
:Re-run script;
if (Run OK?) then (Yes)
note right
Final script is the
Pro-verified version.
end note
else (No)
#ef4444:⚠️ Revert to last
working script;
note right
Pro verification failed.
Use original successful script.
end note
endif
else (No)
:Keep original script;
endif
else (Yes)
:Skip QA (Pro already ran);
endif
}
partition "Final Packaging (Success)" {
#22c55e:-> Create SUCCESS.zip
(logs, script, .bog);
}
else (No)
partition "Final Packaging (Failure)" {
#ef4444:-> Create FAILURE.zip
(logs, last failed script);
}
endif

:Send .zip file to user;

stop

@enduml
