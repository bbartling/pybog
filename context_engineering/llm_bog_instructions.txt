# --- Low-Level XML Structure Rules (For reference/fallback) ---
xml_structure_rules:
  components_tag: p
  root_element:
    name: bajaObjectGraph
    attrs:
      version: "4.0"
      reversibleEncodingKeySource: "none"
      FIPSEnabled: "false"
      reversibleEncodingValidator: "[null.1]="
  folder_structure:
    path: bajaObjectGraph > p{t="b:UnrestrictedFolder" m="b=baja"} > p{name=your_folder t="b:Folder"}
  component_definition:
    tag: p
    required_attributes:
      - n: string  # e.g., "VavDamperMax"
      - t: string  # e.g., "kitControl:Maximum"
      - h: hex_string  # unique within file
      - m: string  # e.g., "kitControl=kitControl"
  component_slots:
    tag: p
    nested_under: component_p
    required_attributes:
      - n: string  # e.g., "in1"
      - v: string  # e.g., "0.0"
  links:
    tag: p
    nested_under: target_component
    required_attributes:
      - n: string  # unique link name
      - t: b:Link
    nested_properties:
      - sourceOrd: "h:<handle>"
      - sourceSlotName: string
      - targetSlotName: string
  layout:
    wsAnnotation:
      tag: p
      name: wsAnnotation
      type: b:WsAnnotation
      value_format: "<x>,<y>,8"

strategy:
  wiresheet_usage:
    simple_logic:
      use_components: [kitControl, control]
      use_case: basic math, boolean, timers, comparisons
      benefit: visual clarity, human readability

  component_slot_strategy:
    - description: "Defines how to configure component slots and actions for different use cases."
    - settable_points:
        - goal: "User can write a value to this point (e.g., a setpoint)."
        - base_configuration:
            - "<p n=\"out\" f=\"s\"/>"
            - "<a n=\"override\" f=\"ho\"/>"
            - "<a n=\"auto\" f=\"ho\"/>"
            - "<a n=\"emergencyOverride\" f=\"h\"/>"
            - "<a n=\"emergencyAuto\" f=\"h\"/>"
        - with_default_value:
            - goal: "To set an initial value for the settable point."
            - note: "This replaces the simple '<p n=\"out\".../>' from the base configuration."
            - configuration:
                - "<p n=\"out\" f=\"s\" t=\"b:StatusNumeric\">"
                - "  <p n=\"value\" v=\"YOUR_VALUE\"/>"
                - "  <p n=\"status\" v=\"0;activeLevel=e:17@control:PriorityLevel\"/>"
                - "</p>"
                - "<p n=\"fallback\" t=\"b:StatusNumeric\">"
                - "  <p n=\"value\" v=\"YOUR_VALUE\"/>"
                - "</p>"
    - readonly_points:
        - goal: "User can only read the value; it is a result of a calculation."
        - configuration:
            - "<p n=\"out\" f=\"h\"/>"
            - "<a n=\"set\" f=\"ho\"/>"
            - "<a n=\"override\" f=\"ho\"/>"
            - "<a n=\"auto\" f=\"ho\"/>"
            - "<a n=\"emergencyOverride\" f=\"h\"/>"
            - "<a n=\"emergencyAuto\" f=\"h\"/>"
  layout:
    strategy_name: Hierarchical Data Flow
    primary_goal: "To arrange components in a left-to-right data flow using strict 10-unit increments."
    increments:
      x_increment: 10
      y_increment: 10
    placement_algorithm:
      - rule: "Default placement: The next block is placed at x_coordinate + x_increment."
      - special_rule_multi_input:
          - condition: "A block (target) receives links from multiple source blocks."
          - action_sources: "Arrange source blocks in a vertical stack to the left of the target. Each subsequent source is placed at y_coordinate + y_increment."
          - action_target: "Position the target block to the right of the sources (x_coordinate + x_increment)."
    typewriter_logic:
      - description: "Handles wrapping when the layout exceeds the horizontal boundary."
      - track_max_y: "The layout process must track the maximum Y-coordinate used by any block in the current row (maxY)."
      - wrap_behavior:
          - if_x_exceeds: 592
          - then_y_is: "maxY + 10" # Use the standard 10-unit vertical increment for a new line.
          - then_x_resets_to: 10
