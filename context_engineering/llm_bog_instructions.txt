core_concepts:
  bog_file:
    format: zip
    contains: file.xml
  dist_file:
    format: zip
    contains: full_station_structure
    target_path: niagara_user_home/stations/STATION_NAME/config.bog/file.xml
  xml_root_element: bajaObjectGraph

xml_structure_rules:
  components_tag: p
  root_element:
    name: bajaObjectGraph
    attrs:
      version: "4.0"
      reversibleEncodingKeySource: "none"
      FIPSEnabled: "false"
      reversibleEncodingValidator: "[null.1]="
  folder_structure:
    path: bajaObjectGraph > p{t="b:UnrestrictedFolder" m="b=baja"} > p{name=your_folder t="b:Folder"}
  component_definition:
    tag: p
    required_attributes:
      - n: string  # e.g., "VavDamperMax"
      - t: string  # e.g., "kitControl:Maximum"
      - h: hex_string  # unique within file
      - m: string  # e.g., "kitControl=kitControl"
  component_slots:
    tag: p
    nested_under: component_p
    required_attributes:
      - n: string  # e.g., "in1"
      - v: string  # e.g., "0.0"
    enum_values:
      rule: use_fully_qualified_type  # e.g., kitControl:TstatAction.direct
  links:
    tag: p
    nested_under: target_component
    required_attributes:
      - n: string  # unique link name
      - t: b:Link
      - m: b=baja
    nested_properties:
      - sourceOrd: "h:<handle>"  # prefix 'h:' is mandatory
      - sourceSlotName: string  # e.g., "out"
      - targetSlotName: string  # e.g., "inA"
  layout:
    wsAnnotation:
      tag: p
      name: wsAnnotation
      type: b:WsAnnotation
      value_format: "<x>,<y>,8"

generating_program_objects:
  critical_rule: >
    LLM must generate ONLY the Java code inside onStart(), onExecute(), and onStop(),
    plus any helper methods at the class level.
  do_not_generate:
    - import_statements
    - class_definition: 'public class ProgramImpl extends ProgramBase'
    - slot_getter_setter_methods: ['getIn1()', 'setOut()', etc.]
  notes:
    - workbench_handles_framework: true
    - helper_methods_must_be_class_level: true
    - use_clock: 'Clock.schedule() for timers'
  xml_example:
    type: xml
    content: |
      <p h="1" n="MyJavaLogic" t="program:Program" m="program=program">
        <p n="execute" v="<![CDATA[
  // Member variables here (if any)
  Clock.Ticket ticket;
  ArrayList<Double> buffer = new ArrayList<>();

  public void onStart() throws Exception {
    updateTimer();
  }

  public void onExecute() throws Exception {
    updateTimer();
    // ... logic ...
  }

  public void onStop() throws Exception {
    if (ticket != null) ticket.cancel();
  }

  // Helper methods here
  void updateTimer() {
    if (ticket != null) ticket.cancel();
    ticket = Clock.schedule(getComponent(), BRelTime.makeSeconds(10), BProgram.execute, null);
  }
        ]]>"/>
        <p n="in1" t="b:StatusNumeric" f="s"/>
        <p n="out1" t="b:StatusNumeric" f="s"/>
      </p>

EXAMPLE 2. Runtime Accumulator (XML Format)

```xml
<p h="2" n="RuntimeAccumulator" t="program:Program" m="program=program">
  <p n="execute" v="<![CDATA[
Clock.Ticket ticket;
long runtimeSeconds = 0;

public void onStart() throws Exception {
  updateTimer();
}

public void onExecute() throws Exception {
  if (getPumpCmd()) runtimeSeconds += 60; // assumes 1-minute schedule
  setRuntimeMinutes(runtimeSeconds / 60.0);
}

public void onStop() throws Exception {
  if (ticket != null) ticket.cancel();
}

void updateTimer() {
  if (ticket != null) ticket.cancel();
  ticket = Clock.schedule(getComponent(), BRelTime.makeMinutes(1), BProgram.execute, null);
}
]]>"/>
  <p n="pumpCmd" t="b:StatusBoolean" f="s"/>
  <p n="runtimeMinutes" t="b:StatusNumeric" f="s"/>
</p>
```

EXAMPLE 3. Lead-Lag Weekly Switch (XML Format)

```xml
<p h="3" n="LeadLagWeekly" t="program:Program" m="program=program">
  <p n="execute" v="<![CDATA[
boolean leadIsA = true;

public void onStart() throws Exception {
  long currentWeek = System.currentTimeMillis() / (7L * 24 * 3600 * 1000);
  leadIsA = (currentWeek % 2 == 0); // even weeks: A leads, odd weeks: B leads
}

public void onExecute() throws Exception {
  boolean enable = getEnable();

  setPumpACmd(enable && leadIsA);
  setPumpBCmd(enable && !leadIsA);
}

public void onStop() throws Exception {
  setPumpACmd(false);
  setPumpBCmd(false);
}
]]>"/>
  <p n="enable" t="b:StatusBoolean" f="s"/>
  <p n="pumpACmd" t="b:StatusBoolean" f="s"/>
  <p n="pumpBCmd" t="b:StatusBoolean" f="s"/>
</p>
```
